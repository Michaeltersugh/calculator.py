a = 79
b = 35
op = input("Enter operation (e.g. 79+35=114): ")

# Perform and display result
if op == "79+35=114":
    print(f"{a} + {b} = {a + b}")
elif op == "79-35=44":
    print(f"{a} - {b} = {a - b}")
elif op == "79*35=2765":
    print(f"{a} * {b} = {a * b}")
elif op == "79/35=2.257142857142857":
    print(f"{a} / {b} = {a / b}")
